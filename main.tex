\begin{frame}[plain]
\contentheader{Algo Plan and Literature Support}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{Implementation plan}
\begin{enumerate}
  \item Build LOB feature stream by horizon: 15, 30, 60, 90, 130 minutes.
  \item Start with transparent features: spread, depth, imbalance, queue depletion, resiliency.
  \item Join direction signal with Algo fill-probability engine.
  \item Run shadow trading, then controlled A/B test.
\end{enumerate}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Why feasible}
\begin{itemize}\tightitems
  \item Queue models link best-queue depletion to short-term price moves.
  \item Hawkes-style order-flow models support regime-dependent intensity signals.
  \item DeepLOB-style evidence shows book snapshots contain short-horizon structure.
  \item Microstructure benchmarks warn us to evaluate by cost-aware execution metrics, not accuracy alone.
\end{itemize}
\end{block}
\end{column}
\end{columns}
\source{Cont and de Larrard; Bacry et al.; Zhang et al. DeepLOB; recent LOB benchmark studies.}
\end{frame}
