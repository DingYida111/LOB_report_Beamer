\begin{column}{0.42\textwidth}
\begin{block}{Case 1: favorable direction}
\begin{itemize}\tightitems
  \item Direction supports the order side.
  \item Expected fill probability is higher.
  \item Execution efficiency: lower slippage and lower market impact.
\end{itemize}
\end{block}
\begin{block}{Case 2: adverse direction}
\begin{itemize}\tightitems
  \item Direction works against the FAK order.
  \item Revise fill expectation and aggressiveness.
  \item Improve success rate and reduce unhedged risk.
\end{itemize}
\end{block}

\begin{column}{0.48\textwidth}
\begin{block}{How to read the state}
\begin{itemize}\tightitems
  \item \key{High capacity + fast resiliency:} raise FAK confidence.
  \item \key{High cost or toxicity:} revise fill expectation and aggression.
  \item \key{High internal fit:} cross internally before external routing.
  \item \key{Weak state:} delay, slice smaller, or hedge earlier.
\end{itemize}
\end{block}
\vspace{0.25em}
\begin{block}{Design rule}
\begin{center}
{\bfseries Price is the observation. Liquidity is the control state.}
\end{center}
\end{block}
